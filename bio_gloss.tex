\newglossaryentry{effektorkomplex}{
    name={Effektorkomplex},
    description={Multimerer Proteinkomplex, der eine regulatorische
    RNA als Guide nutzt, um spezifische Zielmoleküle zu erkennen und
    funktionell zu beeinflussen. Im RNAi-Kontext: RISC mit AGO als
    Kernkomponente; im CRISPR-Kontext: Cas9/sgRNA-Komplex.
    Verbindet Sequenzspezifität (durch die RNA) mit katalytischer
    Aktivität (durch das Protein).},
    type=defi
}

\newglossaryentry{guiderna}{
    name={Guide-RNA},
    description={Einzelsträngige RNA, die einen Effektorkomplex durch
    Basenpaarung zur komplementären Zielsequenz dirigiert. Im RNAi-Kontext:
    der in RISC geladene miRNA- oder siRNA-Strang; im CRISPR-Kontext:
    die sgRNA, die Cas9 zur Ziel-DNA leitet.},
    type=defi
}

\newglossaryentry{prämrna}{
    name={prä-mRNA},
    description={Unreifes, primäres RNA-Pol-II-Transkript im Zellkern

    Bereiche. Wird durch 5'-Capping, \gls{spleißen} (Entfernung der Introns)
    und 3'-Polyadenylierung zur reifen mRNA prozessiert,
    die anschließend ins Zytoplasma exportiert wird.},
    type=defi
}

    translationaler Ebene},
    type=defi
}

\newglossaryentry{dicer}{
    name={Dicer},
    description={Evolutionär konservierte RNase-III-\gls{endonuklease}, die
    lange doppelsträngige RNA (\gls{dsrna}) oder \gls{premirna} in kurze
    21--23\,nt-Fragmente mit charakteristischen 2-nt-3'-Überhängen
    prozessiert (\gls{sirna} bzw.\ \gls{mirna}). Das Produkt wird anschließend
    in den \gls{risc}-Komplex geladen.},
    type=defi
}

\newglossaryentry{risc}{
    name={RISC-Komplex (RNA-induced silencing complex)},
    description={\gls{effektorkomplex} der
    RNA-Interferenz. Kernkomponente ist ein Argonaute-Protein (AGO),
    das einen einzelsträngigen \gls{guiderna}-Strang (\gls{sirna} oder \gls{mirna}) bindet
    und komplementäre Ziel-\glspl{mrna} erkennt. Bei vollständiger Komplementarität
    erfolgt endonukleolytische Spaltung (Slicer-Aktivität von AGO2);
    bei partieller Komplementarität wird die Translation reprimiert
    oder die \gls{mrna} destabilisiert.},
    type=defi
}

\newglossaryentry{mirnaduplex}{
    name={miRNA-miRNA* Duplex},
    description={Kurzlebiges, ca.\ 21--23\,bp langes RNA-Duplex-Intermediat,

    Guide-Strang (\gls{mirna}) wird in \gls{risc} geladen; der komplementäre

\newglossaryentry{ptgs}{
