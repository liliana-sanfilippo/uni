\newglossaryentry{occurrence count}{
name={occurrence count},
description={Let $x \in \sum^m$ and choose $q \in [1, m]$. The occurrence count of a q-gram $z \in \sum^q$ in x is
the number
\[
Occ(x, z) := |{i \in [1, m - q + 1] : x[i . . . i + q - 1] = z}|
\]
},
type=defi
}


\newglossaryentry{q-gram profile}{
name={q-gram profile},
description={Let $\sigma = |\sum|$. The q-gram profile of x is a $\sigma^q$-element vector $p_q(x) := (p_q(x)_z)_{z\in\sum^q}$, indexed by all q-grams, defined by $p_q(x)_z := Occ(x, z)$},
type=defi
}

\newglossaryentry{q-gram distance}{
name={q-gram distance},
description={
The q-gram distance of $x \in \sum^m$ and $y \in \sum^n$ is defined for $1 \leq q \leq min\{m, n\}$ as
\[d_q(x, y) := \sum_{z\in\sum^q}
|p_q(x)_z - p_q(y)_z |
\]},
type=defi
}
