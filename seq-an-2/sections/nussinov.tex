\begin{algorithmic}

\begin{align*}
    M(1,6) & = max \begin{cases}
                       M(2,6) & = 1 \\ % nach rechts
                       M(1,5) & = 2 \\ % nach oben
                       M(2,5) + \mu(1,6) = M(2,5) +  score(\{G, A\}) = 1 + 0 & = 0 \\ % rechts oben + score
                       max_{2 < k < 6} \{ M (1, k - 1) + M (k, 6) \} = * & = 2
    \end{cases} \quad =  2
\end{align*}

\begin{align*}
    *   max_{2 < k < 6} \{ M (1, k - 1) + M (k, 6) \} & = max\{
    M (1,3-1 ) + M (3, 6) ,
    M (1,4-1 ) + M (4, 6) ,
    M (1,5-1 ) + M (5, 6)
    \} \\
    & =  max\{
    M (1,2 ) + M (3, 6) ,
    M (1,3 ) + M (4, 6) ,
    M (1,4 ) + M (5, 6)
    \} \\
    & =  max\{
    0 + 1,
    1 + 0,
    2 + 0
    \}  \\
    & = max \{
    1,
    1 ,
    2
    \}  \\
    & = 2 \quad  mit\ k=5
\end{align*}
