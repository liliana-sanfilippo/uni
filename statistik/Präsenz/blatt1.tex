\begin{align*}
    A_0 & = \{(0,0), (0,1), (1,0), (2,0), (0,2), (0,3), (3,0)\} \\
    A_1 & = \{(1,1)\} \\
    A_2 & = \{(1,2), (2,1)\} \\
    A_3 & = \{(1,3), (3,1)\} \\
    A_4 & = \{(2,2)\} \\
    A_6 & = \{(2,3), (3,2)\} \\
    A_9 & = \{(3,3)\}
\end{align*}
