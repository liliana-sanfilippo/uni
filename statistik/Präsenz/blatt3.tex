\section{Aufgabe 1.1}
\subsubsection{a.}
Seien $\Omega, A, B$ Mengen mit $A \subseteq \Omega, \quad B \subseteq \Omega$. Sei $(\Omega, \mathbb{P})$ ein W-Raum. 
\begin{enumerate}
    \item z.z. $(A \cup B)\backslash (A \cap B) = (A \backslash B) \dot\cup (B \backslash A)$
    \item Leiten Sie aus 1. eine Formel für $P(A\cup B)$ her 
    \item z.z. $\mathbb{P}(\Omega \backslash (A \cup B)) \leq 1 - max \{\mathbb{P}(A), \mathbb{P}(B)\}$
\end{enumerate}
Tipp: Distributivgesetz für Mengen (mehrfach) und $(x\backslash y) = x \cap y^c$
\begin{align*}
    (A \cup B)\backslash (A \cap B) & = (A \cup B)\cap (A \cap B)^c & de Morgan \\
    & = (A \cup B)\cap (A^c \cup B^c) & Distr. \\
    & = (A^c \cap B) \cup (A^c \cap A) \cup B^c & Distr. \\
     & =\textcolor{Orange}{ (A^c \cap (A \cup B)) \cup (B^c \cap (A \cup B))}   \\
     & = \textcolor{Orange}{(A^c \cap A) \cup (B \cap A^c) \cup (A \cap B^c) \cup (B \cap B^c)} \\
      & = (A \cap B^c) \textcolor{Orange}{ \cup } (B \cap A^c) \\
      & = (A \backslash B) \cup (B \backslash A)
\end{align*}
\textcolor{Orange}{Disjunktheit gesondert zeigen: 
\[(B \backslash A) \cap (A \backslash B) = (A \cap B^c) \cup (B \cap A^c) = \emptyset \]}
\subsubsection{b.}
\textcolor{Orange}{
\begin{align*}
 \underbrace{ \underbrace{  \p((A \cup B) \backslash (A \cap B) )}_{\p(A \cup B) - \p (A \cap B) }}_{da A \cap B \subset A \cup B} & = \underbrace{\p ((A \backslash B)\dot\cup (B \backslash A)}_{\p (A \backslash B) + \p(B \backslash A) }  ) \\
 \Rightarrow  \p(A\cup B) & = \p(A \backslash B) + \p (B \backslash A) + \p (A \cap B) 
\end{align*}
}
\subsubsection{c.}
\textcolor{Orange}{
\begin{array}{rl}
     & A \subset A \cup B, \quad B \subset A \cup B \\
    \overset{Monotonie}{\Rightarrow } &  \p(A), \quad \p(B) \leq \p(A \cup B) \\
    \Rightarrow  & max \{\p(A), \p(B) \leq \p(A \cup B) \} \\ \\
    \p((A \cup B)^c) & = \p (\Omega  \backslash (A \cup B) ) \\
    & = 1- \p(A \cup B) \geq 1- max \{\p(A), \p(B)\}
\end{array}
}
