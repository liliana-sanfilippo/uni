\newglossaryentry{stst}{
    name={stückweise stetig},
    description ={Eine Funktion \(f \colon [a,b] \to \mathbb{R}\) heißt \emph{stückweise stetig}, wenn es eine Zerlegung
                      \[
                          a = x_0 < x_1 < \cdots < x_n = b
                      \]
                      gibt, sodass \(f\) auf jedem offenen Teilintervall
                      \[
                          (x_{i-1},x_i), \quad i=1,\dots,n,
                      \]
                      stetig ist und an den Zerlegungspunkten die einseitigen Grenzwerte
                      \[
                          \lim_{x \to x_i^-} f(x)
                          \quad \text{und} \quad
                          \lim_{x \to x_i^+} f(x)
                      \]
                      jeweils existieren und endlich sind.}
}
