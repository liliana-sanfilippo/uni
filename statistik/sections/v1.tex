\subsection*{Begriffe}
\begin{defi}
    \textbf{\Gls{elementarereignis} \ele} \\
    Möglicher Ausgang eines Zufallsexperiments. \\
    Es ist üblich, für Elementarereignisse den kleinen griechischen Buchstaben \ele (omega) zu verwenden.
    Davon weichen wir in konkreten Situationen manchmal ab.
    So schreiben wir zum Beispiel auch $k$, wenn das Elementarereignis eine natürliche Zahl ist, oder $x$, wenn es
    eine reelle Zahl ist.
\end{defi}
\begin{bsp} \labsatz
    \begin{itemize}
        \item Würfeln: $\ele = 3$ (3 gewürfelt)
        \item Zeitmessung: $\ele = 100$ (Sec)
    \end{itemize}
\end{bsp}

\begin{defi}
    \textbf{\Gls{eraum}} $\eraum$ \\
    Die Menge aller Elementarereignisse $\eraum  \neq \emptyset$ ist die Menge aller möglichen Ausgänge
    eines Zufallsexperiments. \\
    Ein Ereignisraum ist also zunächst einmal einfach eine nichtleere Menge.  \\
    Für einen Ereignisraum verwenden immer den großen griechischen Buchstaben $\eraum$ (Omega).
    Wenn wir
    mehrere Ereignisräume benötigen, so schreiben wir
    $\eraum, \tilde {\eraum}, \eraum_1, \eraum_2, \dots$ usw.
\end{defi}
\begin{bsp} \labsatz
    \begin{itemize}
        \item \textbf{Würfeln:}  $\eraum  = \{1,\dots.,6\}$ oder $\eraum  = \{ wuerfelsymbole \}$ \\
        Die Modellierung durch Würfelsymbole zeigt, daß es beim Modellieren wichtig ist anzugeben,
        welche Interpretation unsere Kodierung der Elementarereignisse haben soll.
        Man soll daher
        stets einen Satz der Art
        \begin{center}
            ``Dabei bedeutet $\ele = wuerfelsymbol1$, dass der Würfel eine Eins zeigt''
        \end{center}
        schreiben, wenn man einen Ereignisraum angibt.
        \item \textbf{Zeitmessung:} $\eraum  = [0, \infty)$ ist grundsätzlich eine sinnvolle Wahl, wenn wir das Messen einer
        zufälligen Zeit modellieren wollen.
        Sofern Zusatzinformationen vorliegen, können wir natürlich
        ein kleineres Intervall wählen.
        \kom{Siehe die Klammernnotation, die bedeutet die Null ist drin, aber infinity nicht.
            $\infty$ ist keine normale Zahl, sondern ein Symbol für Unbeschränktheit. Deshalb schreibt man praktisch immer die runde klammer auf der Seite.}
    \end{itemize}
\end{bsp}
\noindent Häufig interessiert uns gar nicht der exakte Ausgang eines Zufallsexperiments, sondern nur, ob der
Ausgang eine bestimmte Eigenschaft hat: Haben wir eine gerade Zahl gewürfelt?
Ist die gemessene
Zeit kürzer als eine Minute?
Wir fassen also jene Elementarereignisse zusammen, die die gewünschte
Eigenschaft haben.
Dies führt zur folgenden Definition.

\begin{defi} \textbf{Ereignis} $A$ \\
(Messbare) Teilmenge des Ereignisraums $A \subseteq \eraum  $(
$A = \emptyset, A = \eraum )$ sind zulässig) \\
Sprechweise: $\ele \in A$ bedeutet ``A ist eingetroffen''
\end{defi}
\noindent Im Fall, dass der Ereignisraum $\eraum \subseteq \rz^n$ weder eine endliche noch eine abzählbar unendliche
Menge ist, beschränken wir uns auf Ereignisse, die Intervalle oder bekannte geometrische Figuren wie Dreieck,
Rechteck oder Kugel sind.
Auch abzählbare Schnitte oder Vereinigungen solcher Objekte sind zulässig, sowie deren abzählbare Schnitte oder Vereinigungen.
Und so weiter.
Diese Einschränkung erlaubt es uns, Fragen der mathematischen Maß- und Integrationstheorie zu vermeiden.

\begin{bem} \labsatz
    \begin{itemize}
        \item Ein Ereignis $A \subseteq \eraum$ ist also ein (meßbares) Element der Potenzmenge $\potm(\eraum)$
        von $\eraum$, das heißt, $A \in \potm(\eraum)$.
        \item Ist $\ele \in \eraum$ ein Elementarereignis, so ist die
        einelementige Menge $\{\ele\} \in \potm(\eraum)$
        ein Ereignis in $\eraum$.
    \end{itemize}
\end{bem}

\begin{bsp} \labsatz
    \begin{itemize}
        \item \textbf{Würfeln} einer geraden Zahl: \[A = \{2,4,6\}, \quad \ele = 4 \Rightarrow \ele \in A\] Wir sagen: ``gerade Zahl gewürfelt''
        \item \textbf{Zeitmessung:} $A = [100, 200] \subset \eraum  = [0,\infty)$
    \end{itemize}
\end{bsp}


\section{Vorgehen beim Modellieren}\label{sec:vorgehen-beim-modellieren}
\kom{Beim geeigneten $\eraum $ können Modelle sich bereits unterscheiden. Bspw. Münzwurf: Man kann ``k'', ``z'' nahmen oder ``Kopf'', ``Zahl''. Das sind verschiedenen $\eraum $, aber beide wären passend.}
\begin{itemize}
    \item Finden des geeigneten $\eraum $
    \item Ereignisse müssen \gls{ws} zugeordnet bekommen
    \begin{enumerate}
        \item \textbf{Möglichkeit}: \gls{ws} für $\eraum $ abzählbar
    \end{enumerate}
\end{itemize} \vskip 14pt \orangenote{Wenn sie abzählbar sagt, meint sie sowohl endlich als auch abzählbar unendlich.}

\begin{defi} \label{def:zaehldichte}
    \textbf{Zähldichte / \acrfull{wfk}} \\
    Sei $\eraum  \neq \emptyset$ eine abzählbare Menge.
    \begin{itemize}
        \item[(i)] Eine Abbildung $\wfk : \eraum  \rightarrow [0,1]$ heißt mit \[
                                                                                                         \underbrace{\sum_{\ele \in \eraum } \wfk (\ele) = 1}_{\text{\txo{Normierung}}}
        \] \underline{Zähldichte / \gls{wfk}} auf $\eraum $. \\
        \kom{Die Reihe (Normierung) muss 1 sein und konvergieren.} \kom{Wenn man max abzählbar hat, dann sind die Wahrscheinlichkeiten einfach teile der Summe (Normierung) und daher ist das einfach.}
        \item[(ii)] Die \gls{ws} $\pe(A)$ einer Ereignisses $A \subseteq \eraum $ ist bei gegebener \gls{wfk} $\wfk$ definiert durch
        \[ \pe(A)\coloneqq \sum_{w \in A} \wfk(\ele)\]   \kom{Was hat die Potenzmenge von Omega damit zu tun?}
        \noindent $\pe$ heißt \gls{wmaß} und liegt immer zwischen Null und 1
    \end{itemize}
    \begin{bem}\label{bem:konvention}
        \textit{Konvention}: $\pe(\emptyset) = 0$
    \end{bem}
    \begin{bem}
        \[0 \leq \pe \leq 1, \quad \forall A \subseteq \eraum  \quad \txo{\underbrace{(\forall A \in \potm(\eraum )) }_{\potm(\eraum ) \text{Potenzmenge von } \eraum }} \]
        $\pe$ ist eine Abb.
        \[
            \pe: \potm(\eraum ) \rightarrow [0,1] \quad  \text{ mit }  \pe(\emptyset) = 0, \quad  \pe(\eraum ) = 1
        \]
    \end{bem}
\end{defi}

\begin{lem} \textbf{Eigenschaften des Wahrscheinlichkeitsmaßes für abzählbare Ereignisräume} \\
    Sei $\eraum \neq \emptyset$ eine abzählbare Menge und $\pe$ ein Wahrscheinlichkeitsmaß auf $\eraum$.
Dann gelten:
    \begin{itemize}
        \item $0 \leq \pe (A) \leq 1$ für alle $A \leq \potm(\eraum)$
        \item $\pe(\emptyset) = 0$
        \item $\pe(\eraum) = 1$
    \end{itemize}
    \paragraph{Beweis.}
    \begin{itemize}
        \item Untere Schranke: Da $\pe(A) = \sum_{\wfk \in A} \wfk(\ele)$ gilt und alle Summanden nichtnegativ sind,
        folgt $\pe(A) \geq 0$
        \item Obere Schranke: Aus
        \[
            \pe(A) = \sum_{\ele \in A} \wfk(\ele) \leq \sum_{\ele \in \eraum} \wfk(\ele) = 1
        \]
        folgt $\pe(A) \leq 1$.
        Dabei haben wir in der Abschätzung erneuet verwendet, daß alle Summanden
        nichtnegativ sind.
        \item $\pe (\emptyset) = 0$ gilt gemäß Konvention aus \autoref{bem:konvention}.
        \item $\pe(\eraum) = \sum_{\ele \in \eraum} \wfk(\ele) = 1$ folgt direkt aus der Normierung der Wahrscheinlichkeitsfunktion.
    \end{itemize}
\end{lem}

\begin{defi}
    \textbf{\gls{wraum}} \\
    $\underbrace{\wraum}_{\txo{(\eraum , \wfk)}}$ heißt \underline{\gls{wraum}}.
    \begin{bem}
        Für $\eraum $ abzählbar: $\wfk$ legt $\pe$ eindeutig und umgekehrt!
        \[P(A)=\sum_{\ele \in A} \wfk (\ele) \quad \forall A \in \potm(\eraum )\]
        \[\wfk (\ele) = \pe(\underbrace{\{\ele\}}_{\txo{\text{1-elementige Menge}}}) \quad \forall \ele \in \eraum \]
        \paragraph{Länger aus dem Skript:} Da es eine ein-eindeutige Zuordnung zwischen der
        Wahrscheinlichkeitsfunktion $\wfk$ und dem Wahrscheinlichkeitsmaß $\pe$ gibt, bezeichnen wir auch das Paar $\wraum$
        als Wahrscheinlichkeitsraum.
        Die Zuordnung von $\wfk$ zu $\pe$ ist direkt durch die Definition von $\pe$ in Definition
        \autoref{def:zaehldichte} gegeben.
        Für eine
        gegebene Wahrscheinlichkeitsfunktion erhalten wir so ein eindeutig bestimmtes Wahrscheinlichkeitsmaß.
        Ist ein Wahrscheinlichkeitsmaß $\pe$ gegeben, so ergibt sich die zugehörige Wahrscheinlichkeitsfunktion $\wfk$
        sofort aus den Wahrscheinlichkeiten der Elementarereignisse:
        \[
            \wfk(\ele) = \pe(\{\ele\}), \faoio
        \]
        Beachten Sie hier, dass die Wahrscheinlichkeitsfunktion $\wfk$ als Argument ein Elementarereignis, also
        ein einzelnes $\oio$ hat, während das Wahrscheinlichkeitsmaß $\pe$ auf der Potenzmenge $\pe(\eraum)$ von $\eraum$
        definiert ist und daher das einelementige Ereignis $\{\ele\}$ als Argument hat.
    \end{bem}
\end{defi} %

\begin{defi}
    \textbf{Gleichverteilung} \\
    Sei $\eraum  \neq \emptyset$ eine endl.
    Menge, d.h. $0 < \#\eraum  < \infty$.
    Gilt $\wfk(\ele) = \wfk(\tilde{\ele}) \quad  \forall \ele, \tilde{\ele} \in \eraum $ d.h.\ die \gls{wfk} $\wfk$ ist konstant, so heißen die wfkt mu und das zugehörige wmaß P die \underline{Gleichverteilung auf $\eraum $}
\end{defi}
\begin{lem}
    Sei $\wfk$ die Gleichverteilung auf $\eraum $.
    Dann gelten:
    \begin{itemize}
        \item[(a)] $\wfk(\ele) = \frac{1}{\#\eraum } \quad \forall \ele \in \eraum $
        \item[(b)] $\pe(A) = \frac{\#A}{\#\eraum } \forall A \in \pe(\eraum )$ \txo{($\forall A \subseteq \eraum $)}
    \end{itemize}

    \paragraph{Beweis.}
    \begin{itemize}
        \item[(a)] \(1 = \sum_{\ele \in \eraum } \wfk(\ele) = \) \footnote{Wir dürfen annehmen, dass ein $c \in [0,1]$ existiert mit $\wfk(\ele) = c \quad \forall \ele \in \eraum $}$\ c \cdot \# \eraum  \Rightarrow c = \frac{1}{\#\eraum }$
        \item[(b)] $\pe(A) = \sum_{\ele \in A } \wfk(\ele) \overset{(a)}{=} \frac{\#A}{\# \eraum }$ \\
        \txo{$\pe(A)$ ist die Anzahl der günstigen Ausgänge, geteilt durch die Anzahl aller möglichen Ausgänge} \\

    \end{itemize}
\end{lem}
\begin{bem}
    Im Fall der Gleichverteilung reduziert sich das Berechnen von WS auf Abzählprobleme. \\ \kom{Mathe 1}
\end{bem}
\begin{bem} % aus Skript
    \textbf{Keine Gleichverteilung, wenn der Ereignisraum nicht endlich ist} \\
    Warum müssen wir voraussetzen, dass $\# \eraum < \infty $?
    Nehmen wir an, es gäbe eine Gleichverteilung $\wfk$ auf einer abzählbar unendlichen Menge
    $\eraum$.
    Wir können der Einfachheit halber annehmen, dass $\eraum$ die Menge
    der natürlichen Zahlen $\mathbb{N}$ ist.
    Dann gibt es also ein $c \in [0, 1]$ mit
    \[
        1 = \sum^{\infty}_{k = 1} \wfk(k)    = c \cdot \# \mathbb{N}
    \]
    Nun gibt es zwei Fälle.
    Entweder, $c = 0$.
    Dann erhalten wir den Widerspruch $1 = 0$.
    Andernfalls, wenn $c > 0$ gilt, erhalten wir ebenfalls einen Widerspruch, nämlich $1 = \infty$.
    Folglich kann es auf einer nicht endlichen Menge keine Gleichverteilung geben
\end{bem}
