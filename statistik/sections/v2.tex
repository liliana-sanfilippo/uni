\begin{defi} \label{def:dichte}

    EIne Funktion $f: \mathbb{R} \rightarrow [0, \infty)$, die (uneigentlich) Riemann-integrierbar ist, heißt
    \underline{\dichte}, sofern
    \[
        \int_{- \infty}^{\infty} f (x) dx = 1
    \]
\end{defi}
\begin{bem} \defirefman{dichte}
\begin{itemize}
    \item  % a
    \item $f$ monoton oder $f$ stetig $\Rightarrow \int_a^b f(x) dx$ existiert für alle $(a, b \in \mathbb{R})$
    \txc{auch: für $- \infty < a < b < \infty$}
    % b
    \item $f$ stückweise stetig oder stückweise monoton, $\Rightarrow \int_a^b f(x) dx$ existiert $(a, b \in \mathbb{R})$

\begin{defi}
    \textbf{Uniforme Verteilung} (Kontinuumsäquivalent zu gleichverteilt) \\
    Die uniforme Verteilung auf einem beschränkten Intervall $I \subseteq \mathbb{R}$ ist gegeben durch eine \dichte
    \[
        \dcht(x) = \begin{cases}
                   \frac{1}{\mid I \mid } & x \in I \\ 0 & sonst
        \end{cases}
    \]
    wobei $\mid I \mid $ die Intervalllänge bezeichnet.
\end{defi}
\begin{defi}
    \textbf{Exponentialverteilung} \label{defi:exponentialverteilung} \\
    Die Exponentialverteilung zum Parameter $\alpha > 0$ ist gegeben durch die \dichte
    \begin{equation}
        \label{eq:Exponentialverteilung}
        \dcht(x) = \begin{cases}
                   0 & \text{ für } x < 0 \\
                   \alpha e ^{- \alpha x} &  \text{ für } x \geq 0
        \end{cases}
    \end{equation}
\end{defi}

\begin{satz}
    Seien $A, B, A_1, A_2, \dots \subset \eraum $ (und Intervalle, falls $\pe$ durch \dichte gegeben ist)
    \begin{itemize}
        \item[(a)] Additivität: Seien $A_1, A_2, \dots$ paarweise disjunkt, dann gilt:
        \[\pe(\underset{i}{\dot\cup} A_i) = \sum_i \pe(A_i)\]
        Gilt auch für \Glspl{dichte}! \\
        \textbf{Beweis.}  \\
        \begin{align*}
            \pe(\underset{i}{\dot\cup} A_i)&  = \pe(\{\glssymbol{elementarereignis} \in \eraum : \exists i \text{ mit } \glssymbol{elementarereignis} \in A_i\}) \\
            & = \sum_i \txo{\underbrace{\textcolor{black}{\sum_{\glssymbol{elementarereignis} \in A_i}}}_{\pe(A_i)}} \glssymbol{wfk} (\glssymbol{elementarereignis}) \\
            & = \sum_i \pe(A_i)
        \end{align*}
        \orangenote{Wir haben die Summe/Reihe umsortiert}

        \item[(b)] Komplementärwahrscheinlichkeit
        \[\pe(A^c) = \pe(\eraum  \backslash A) = 1 - \pe(A)\]
        \textbf{Beweis.}
        \begin{align*}
            \eraum  =  A \dot\cup A^c & \overset{(a)}{\Rightarrow} 1 \overset{\txo{Normierung}}{=} \pe(\eraum ) = \pe( A \dot\cup A^c) \overset{(a)}{=}\pe(A) + \pe(A^c) \\
            & \Rightarrow \pe(A^c) = 1-  \pe(A)
        \end{align*}
        \item[(c)] Monotonie: \[A \subset B \Rightarrow \pe(A) \leq \pe(B)\]
        \textbf{Beweis}
        \begin{align*}
            B & = A \dot\cup (B\backslash A) \\
            \pe(B) & = \pe(A) + \txo{
                \underbrace{\textcolor{black}{\pe(B \backslash A) }}_{\geq 0}
            } \geq \pe(A)
        \end{align*}
        \item[(d)] $A \subset B\Rightarrow \pe(A) = \pe(B) - \pe(B \backslash A) $
        \item[(e)] $\pe(A \cup B) = \pe(A) + \pe(B) - \pe(A \cap B)$ \\
        \textbf{Beweis.}
        \begin{align*}
            A \cup B & = A \dot\cup [B \backslash (A \cap B)] \\
            \pe(A \cup B) & = \pe(A) +
            \txo{
                \underbrace{
                    \textcolor{black}{\pe(B \backslash (A \cap B) )}
                }_{ \overset{(d)}{=}\pe(B) - \pe(A \cap B) }
            } \\
            & = \pe(A) + \pe(B) - \pe(A \cap B)
        \end{align*}
    \end{itemize}
\end{satz}
\begin{bem}
    Sind $A, B$ disjunkt, so gilt nach (a): $\pe(A \dot\cup B)  = \pe(A) + \pe(B) $. \\
    Sind $A, B$ nicht disjunkt, so gilt nach (e): $\pe(A \cup B) = \pe(A) + \pe(B) - \pe(A \cap B)$. \\
    Sind $A, B, C$ paarweise disjunkt, so gilt nach (a): $\pe(A \cup B \cup C) \overset{(a)}{=} \pe(A) + \pe(B) + \pe(C) $. \\ \\
    Was passiert, wenn $A,B,C$ nicht paarweise disjunkt sind? \begin{align*}
                                                                  \pe(A \cup B \cup C) & = \pe(A \cup (B \cup C)) \\
                                                                  & = \pe(A) \pe( \cup B \cup C) -  \pe(A \cap (B \cup C)) \\
                                                                  & = \pe(A) + [\pe(B) + \pe(C) - \pe(B \cap C)] - \pe((A \cap B) \cup (A \cap C))\\
                                                                  & = \pe(A) + \pe(B) + \pe(C)  & \text{Mengen }  \\
                                                                  & \quad - \pe(A \cap B) - \pe(A \cap C) - \pe(B \cap C) & \text{Paare von Mengen }  \\
                                                                  & \quad + \pe(
                                                                  \txo{
                                                                      \underbrace{
                                                                          \textcolor{black}{(A \cap B) \cap (A \cap C)}
                                                                      }_{= A \cap B \cap C}
                                                                  }
                                                                  ) & \text{Tripel von Mengen }
    \end{align*}
\end{bem}
\begin{satz}
    \textbf{Ein- und Ausschlussprinzip} \\
    \[A_2,\dots, A_n \subset \eraum  \]
    \[\pe( \bigcup^n_{i=1} A_i ) = \sum^n_{k = 1} (-1)^{k+1} \sum_{a \leq i_1 \leq \dots \leq i_k \leq k} \pe(A_{i_1} \cap \dots \cap A_{i_k}) \]
\end{satz}
