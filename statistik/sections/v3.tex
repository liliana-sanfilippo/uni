\begin{bsp}
    \textbf{Einmaliges Würfeln}  \txc{(2025+2026 zsm.)} \\
    Wie wahrscheinlich ist es, daß eine 1 oder eine 6 gewürfelt wurde?
    Einmaliges Würfeln

    Wie wahrscheinlich ist es, dass eine 1 oder eine 6 gewürfelt wurde?

    $|\eraum| = \{1, \dots, 6\}, \; \pe$ Gleichverteilung, $A = \{1,6\}, \; \pe(A) = \frac{|A|}{|\eraum|} = \frac{2}{6} = \frac{1}{3}$

    Wie wahrscheinlich ist es, dass eine 1 oder eine 6 gewürfelt wurde, wenn wir schon wissen, dass die geworfene Zahl durch 3 teilbar ist?

    Was ist ``$\pe$ von A, gegeben B'' \txc{$\pe(A \mid B)$} mit $B = \{3,6\}$?

    Wahrscheinlichkeit von $A$:

    \[
        \pe(A) = \frac{\text{Anz. günstige Ausgänge}}{\text{Anz. mögliche Ausgänge}} = \frac{|A|}{|\eraum|}
    \]

    Wahrscheinlichkeit von $A$ gegeben $B$:

    \[
        \pe(A \mid B) =
        \frac{\text{Anz. günstige Ausgänge in } A \cap B}{\text{Anz. mögliche Ausgänge in } B}
        = \frac{|A \cap B|}{|B|}
        = \frac{|\{6\}|}{|\{3,6\}|}
        = \frac{1}{2}
    \]
\end{bsp}
\begin{notation}  \txc{(2025+2026 zsm.)} \\
    $|A| = \# A$
\end{notation}
\begin{defi}
    \label{def:bedingte-ws} \textbf{\gls{bws}}   \txc{(2025+2026 zsm.)}\\
    Für Ereignisse A und B heißt
    \[
        \pe(A \mid B) =
        \begin{cases}
            \frac{\pe(A \cap B)}{\pe(B)}, & \text{falls } \pe(B) > 0, \\
            0, & \text{falls } \pe(B) = 0,
        \end{cases}
    \]
    die bedingte Wahrscheinlichkeit von A gegeben B\@.
\end{defi}
\komm{\defirefman{bedingte-ws}: Wenn
    $\pe$ durch eine Dichte gegeben ist, muss man die Definition über Integrale interpretieren.}

\begin{beo}  \txc{(2025+2026 zsm.)} \\
    Bei Gleichverteilung gilt

    \[
        \pe(A \mid B)
        =
        \frac{\frac{\#|A \cap B|}{\#|\eraum|}}{\frac{\#|B|}{\#|\eraum|}} =
        \frac{|A \cap B|}{|B|}
    \]

    D.h., die bedingte Wahrscheinlichkeit von A gegeben B ist tatsächlich der relative Anteil von A in B\@.
\end{beo}
\begin{fr} \txc{(2025+2026 zsm.)} \\
In obigem Beispiel gilt $\pe(A \mid B) > \pe(A)$.
    Gilt das immer?

    \paragraph{Antwort} Nein!
    Sowohl $\pe(A_1 \mid B_1) > \pe(A_1)$ als auch $\pe(A_2 \mid B_2) < \pe(A_2)$ sind möglich.

\end{fr}
\begin{bsp} \txc{(2025+2026 zsm.)}
\[
        \eraum = \{1,\dots,6\} \text{ mit Gleichverteilung } \pe
    \]

    \[
        A_1 = \{1,6\}, \quad A_2 = \{1,2,3,4\}, \quad B = B_1 = B_2 = \{3,6\}
    \]

    Dann gelten

    \[
        \pe(A_1 \mid B) = \frac{|A_1 \cap B|}{|B|} = \frac{1}{2} > \frac{1}{3} = \pe(A_1)
    \]

    \[
        \pe(A_2 \mid B) = \frac{|A_2 \cap B|}{|B|} = \frac{1}{2} < \frac{2}{3} = \pe(A_2)
    \]
\end{bsp}

\begin{reg}
    \textbf{Rechenregeln für bedingte Wahrscheinlichkeiten \txo{(diskreter Fall - analog für Fall einer Dichte)}} \txc{(2025+2026 zsm.)}  \\
    Für ein festes Ereignis $B$ mit $\pe(B) > 0$ definiert

    \[
        \wfk_B(\ele) = \pe(\{\omega\} \mid B) =
        \begin{cases}
            \frac{\pe(\{\omega\})}{\pe(B)} = \frac{\wfk(\{\omega\})}{\pe(B)}, & \text{falls } \omega \in B, \\
            0, & \text{falls } \omega \notin B,
        \end{cases}
    \]

    eine Wahrscheinlichkeitsfunktion, und es gilt

    \[
        \sum_{\omega \in A}\wfk_B(\omega)
        =
        \frac{1}{\pe(B)}
        =
        \sum_{\omega \in A \cap B} \wfk(\omega)
        =
        \frac{\pe(A \cap B)}{\pe(B)}
        =
        \pe(A \mid B)
    \]

    für alle $A \subseteq \eraum$.

    Die Abbildung $A \mapsto \pe(A \mid B)$ ist also ein Wahrscheinlichkeitsmaß.

    Es gelten also alle bekannten Rechenregeln für Wahrscheinlichkeiten, z.B\@.

    \[
        \pe(A^c \mid B) = 1 - \pe(A \mid B)
    \]

    oder

    \[
        \pe(A_1 \cup^+ A_2 \mid B) = \pe(A_1 \mid B) + \pe(A_2 \mid B)
    \]

    \txc{(Achtung: nur für disjunkte Ereignisse.)}

    Achtung!
    Die Abbildung $B \mapsto \pe(A \mid B)$ ist kein Wahrscheinlichkeitsmaß!

    (Beweisen Sie dies selbständig, indem Sie z.B.\ Beispiele finden mit
    $\pe(A \mid B) \neq 1 - \pe(A \mid B^c)$ oder
    $\pe(A\mid B_1 \cup B_2) \neq \pe(A \mid B_1) + \pe(A \mid B_2)$.)
\end{reg}
\begin{bsp}
    \textbf{Gedächtnislosigkeit der Exponentialverteilung} \txc{(2025+2026 zsm.)}  \\
    Parameter $z > 0$; \dichte:
    \[
        \dcht(x) = \begin{cases}
                       z e^{-zx}, & x \geq 0 \\ 0, & sonst
        \end{cases}
    \]
    $ \pe(I) = \intab \dcht(x) \dx $ für $I = (a,b] \subset [0,\infty)$ \\
    Warteschleife beim Kundenservice als Exponentialverteilung
    \begin{itemize}
        \item Sie haben schon 10 Min gewartet $(t=10)$
        \item \gls{ws}, dass Sie noch \textbf{mindestens} weitere 10 Minuten warten? \\ ($T=t+10=20$)
    \end{itemize}
    Gesucht: $\pe([T,a) | [t, \infty) )$
    \[
        \pe([T,a) | [t, \infty) ) = \frac{\pe([T,a)) \cap \pe([t, \infty) )}{\pe([t, \infty) )} =
        \frac{\pe([T, \infty) )}{\pe([t, \infty) )}
    \]
    Nebenrechnung:
    \begin{align*}
        \pe([\tau, \infty) ) &= \int_{\tau}^{\infty} z e^{-zx} \dx \\
        & = \lim_{N \rightarrow \infty } \int_{\tau}^{\infty} z e^{-zx} \dx \\
        & = \lim_{N \rightarrow \infty } ( - e^{-zx}) \mid_{x=\tau}^{N} \\
        & =\lim_{N \rightarrow \infty } [ \underbrace{- e^{-zN}}_{\rightarrow_{N \rightarrow \infty} 0} + e^{-z\tau} ]  \\
        & = e^{-z\tau}
    \end{align*}
    \txc{Zurück zur Hauptrechnung:}
    \begin{align*}
        \pe([T,a) | [t, \infty) ) & = \frac{\pe([T, \infty) )}{\pe([t, \infty) )} \\
        & \overset{NR}{=} \frac{e^{-zT}}{e^{-zt}} \\
        & = e^{-z(T-t)} \\
        & \overset{T=10+t}{=} e^{-zt} \\
        & \overset{NR}{=} \pe([t,\infty))
    \end{align*}
\end{bsp}


\begin{reg}
    \textbf{Neue Rechenregel: Sukzessives Bedingen} \txc{(2025+2026 zsm.)}  \\
    Aus der Definition der bedingten Wahrscheinlichkeit folgt

    \[
        \pe(A \cap B) = \pe(A \mid B)\,\pe(B)
        \qquad \forall\, A,B \subseteq \eraum.
    \]

    Da Multiplikation mit Null wieder Null ergibt, gilt diese Formel auch im Fall
    \(\pe(B)=0\).
    \txo{
        \[
            \pe(B)=0 \Rightarrow \pe(A \cap B) \leq \pe(B) = 0 \Rightarrow \pe(A \cap B) = 0
        \]
    }

    Wiederholtes Anwenden zeigt

    \[
        \pe(A_1 \cap A_2 \cap A_3)
        =
        \pe(A_3 \mid A_1 \cap A_2)\,\pe(A_1 \cap A_2)
    \]

    und damit

    \[
        \pe(A_1 \cap A_2 \cap A_3)
        =
        \pe(A_3 \mid A_1 \cap A_2)\,
        \pe(A_2 \mid A_1)\,
        \pe(A_1)
    \]

    für alle \(A_1, A_2, A_3 \subseteq \eraum\).

    Mit vollständiger Induktion beweist man die allgemeine Formel

    \[
        \pe(A_1 \cap A_2 \cap \cdots \cap A_n)
        =
        \pe(A_n \mid A_1 \cap \cdots \cap A_{n-1})
        \cdots
        \pe(A_2 \mid A_1)\,
        \pe(A_1)
    \]

    für alle \(A_1,\dots,A_n \subseteq \eraum\).
\end{reg}

    \textbf{Wie wahrscheinlich ist es, dass beim Skat jeder der drei Spieler genau ein As hat?} \txc{(2025+2026 zsm.)}

    \paragraph{Modell}
    \[
        \eraum
        =
        \Bigl\{
        (S_1,S_2,S_3)
        :
        S_i \subseteq \{1,\dots,32\},
        \ |S_i|=10
        \ \forall i\in\{1,2,3\},
        \ S_i \cap S_j = \varnothing \text{ für } i\neq j
        \Bigr\}.
    \]

    \((S_1,S_2,S_3)\in\eraum\) bedeute:
    \begin{itemize}
        \item Spieler \(i\) erhält die 10 Karten \(S_i\), \(i\in\{1,2,3\}\).
        \item Die übrigen 2 Karten
        \[
            \{1,\dots,32\}\setminus(S_1\cup^+ S_2\cup^+ S_3)
        \]
        liegen im Skat.
        \item Die vier Asse seien die Karten mit den Nummern \(1,2,3,4\).
        \item Gleichverteilung:
        \[
            \wfk(\omega)
            =
            \frac{1}{|\eraum|}
            =
            \frac{1}
            {\binom{32}{10}\binom{22}{10}\binom{12}{10}}
            \qquad \forall\, \omega\in\eraum.
        \]
    \end{itemize}

    \paragraph{Rechnung}
    Beispiel: Wie wahrscheinlich ist es, dass beim Skat alle genau ein As haben?


    \[
        A_i=\bigl\{|S_i\cap\{1,2,3,4\}|=1\bigr\},
        \qquad i=1,2,3
    \]

    (Spieler \(i\) hat genau ein As)

    \[
        A=A_1\cap A_2\cap A_3
    \]

    (Jeder Spieler hat genau ein As)

    \[
        \pe(A)
        =
        \pe(A_1\cap A_2\cap A_3)
        =
        \pe(A_1)\,\pe(A_2\mid A_1)\,\pe(A_3\mid A_1\cap A_2) \txo{= \pe(A_3\mid A_1\cap A_2)\,\pe(A_2\mid A_1)\,\pe(A_1)}
    \]

    \[
        \pe(A_1)
        =
        \frac{|A_1|}{|\eraum|}
        =
        \frac{\binom{4}{1}\binom{28}{9}\textcolor{red}{\binom{22}{10}\binom{12}{10}}}
        {\binom{32}{10}\textcolor{red}{\binom{22}{10}\binom{12}{10}}}
        =
        \frac{\binom{4}{1}\binom{28}{9}}
        {\binom{32}{10}}
    \]

    \[
        \pe(A_2\mid A_1)
        =
        \frac{|A_1\cap A_2|}{|A_1|}
        =
        \frac{\textcolor{red}{\binom{4}{1}\binom{28}{9}}\binom{3}{1}\binom{19}{9}\textcolor{red}{\binom{12}{10}}}
        {\textcolor{red}{\binom{4}{1}\binom{28}{9}}\binom{22}{10}\textcolor{red}{\binom{12}{10}}}
        =
        \frac{\binom{3}{1}\binom{19}{9}}
        {\binom{22}{10}}
    \]

    \[
        \pe(A_3\mid A_1\cap A_2)
        =
        \frac{|A_1\cap A_2\cap A_3|}
        {|A_1\cap A_2|}
        = \dots =
        \frac{\binom{2}{1}\binom{10}{9}}
        {\binom{12}{10}}
    \]

    Somit

    \[
        \pe(A)
        =
        \frac{\binom{4}{1}\binom{28}{9}}
        {\binom{32}{10}}
        \cdot
        \frac{\binom{3}{1}\binom{19}{9}}
        {\binom{22}{10}}
        \cdot
        \frac{\binom{2}{1}\binom{10}{9}}
        {\binom{12}{10}}
    \]

    \[
        \pe(A)\approx 0.056.
    \]

\begin{defi}
    \textbf{Formel von der totalen Wahrscheinlichkeit} \txc{(2025+2026 zsm.)}  \\
    Sei \((\eraum,\pe)\) ein Wahrscheinlichkeitsraum und \txo{(abzählbar viele $B_i$)}
    \(B_1,B_2,\dots \subseteq \eraum\) \txo{paarweise disjunkte} Ereignisse mit

    \[
        \bigcup^+_i B_i = \eraum.
    \]

    Dann gilt

    \[
        \pe(A)
        =
        \sum_i \pe(A\mid B_i)\,\pe(B_i)
    \]

    für alle Ereignisse \(A \subseteq \eraum\).

    \txc{Hinweis: \(B_1,B_2,\dots\) ist eine Partition von $\eraum$}

    \paragraph{Beweis (im einfachsten Fall).}

    Für

    \[
        \eraum = B_1 \cup^+ B_2\txc{,
            \qquad
            B_1 \cap B_2 = \varnothing}
    \]

    (also \(B_1=B\) und \(B_2=B^c\)) gilt

    \[
        A
        =
        A \cap (B_1 \cup^+ B_2)
        =
        (A\cap B_1)\cup^+(A\cap B_2).
    \]

    Da die Ereignisse \(A\cap B_1\) und \(A\cap B_2\) disjunkt sind, folgt (über Rechenregeln für Wahrscheinlichkeiten)

    \[
        \pe(A)
        =
        \pe(A\cap B_1)+\pe(A\cap B_2)
        \tag{1}
    \]

    und mit der Definition der bedingten Wahrscheinlichkeit

    \[
        \pe(A\cap B_i)
        =
        \pe(A\mid B_i)\,\pe(B_i),
        \qquad i=1,2.
        \tag{2}
    \]

    Einsetzen von (2) in (1) ergibt

    \[
        \pe(A)
        =
        \pe(A\mid B_1)\,\pe(B_1)
        +
        \pe(A\mid B_2)\,\pe(B_2).
    \]

    Der allgemeine Fall geht vollkommen analog und verbleibt als Hausaufgabe.
\end{defi}
\begin{bsp}
    \textbf{Herr Zimperlich und sein Regenschirm} \txc{(2025+2026 zsm.)}  \\
    Herr Zimperlich hasst es, nass zu werden.
    Wenn der Wetterbericht Regen ankündigt, nimmt er seinen Schirm mit einer Wahrscheinlichkeit von \(90\%\) mit.
    (Die fehlenden \(10\%\) sind seiner Vergesslichkeit geschuldet.)

    Auch wenn der Wetterbericht einen trockenen Tag verheißt, nimmt Herr Zimperlich mit einer Wahrscheinlichkeit von \(30\%\) trotzdem seinen Schirm mit.

    Dem feuchten Bielefelder Klima gemäß nehmen wir an, dass für \(60\%\) aller Tage Regen vorhergesagt wird.

    Frage: Wie groß ist die Wahrscheinlichkeit, dass Herr Zimperlich das Haus ohne Schirm verlässt?

    \paragraph{Modellierung}

    \[
        \eraum = \{r,\bar r\}\times\{s,\bar s\}
        = \{(r,s),(\bar r,s),(r,\bar s),(\bar r,\bar s)\}
    \]

    Dabei sei \((r,\bar s)\) das Elementarereignis, dass Regen vorhergesagt wurde und Herr Zimperlich keinen Schirm mitnimmt, usw.

    \[
        R=\{(r,s),(r,\bar s)\}
        \qquad\text{(Regen vorhergesagt)}
    \]

    \[
        S=\{(r,s),(\bar r,s)\}
        \qquad\text{(Schirm dabei)}
    \]

    \[
        \pe(S\mid R)=\frac{9}{10},
        \qquad
        \pe(S\mid R^c)=\frac{3}{10},
        \qquad
        \pe(R)=\frac{6}{10}.
    \]

    \paragraph{Gesucht:}

    \[
        \pe(S^c).
    \]

    \paragraph{Problem:} Wie können wir diese „absolute“ Wahrscheinlichkeit aus den bedingten Wahrscheinlichkeiten bestimmen?

    \paragraph{Lösung:}

    Die Formel der totalen Wahrscheinlichkeit mit

    \[
        \eraum = R \cup R^c
    \]

    zeigt:

    \[
        \pe(S^c)
        =
        \pe(S^c\mid R)\,\pe(R)
        +
        \pe(S^c\mid R^c)\,\pe(R^c).
    \]

    Nach Zurückführen auf die gegebenen Daten und Einsetzen erhalten wir

    \[
        \pe(S^c)
        =
        \bigl[1-\pe(S\mid R)\bigr]\pe(R)
        +
        \bigl[1-\pe(S\mid R^c)\bigr]\bigl[1-\pe(R)\bigr]
    \]

    \[
        =
        \left(1-\frac{9}{10}\right)\frac{6}{10}
        +
        \left(1-\frac{3}{10}\right)\left(1-\frac{6}{10}\right)
        =
        \frac{34}{100}.
    \]
\end{bsp}

\begin{satz}
    \textbf{Satz von Bayes (einfache Version)} \txc{(2025+2026 zsm.)}  \\
    Seien \((\eraum,\wfk)\) ein Wahrscheinlichkeitsraum und
    \(A,B \subseteq \eraum\) Ereignisse mit \(\pe(B)>0\).

    Dann gilt

    \[
        \pe(A\mid B)
        =
        \pe(B\mid A)\cdot\frac{\pe(A)}{\pe(B)}.
    \]

    \paragraph{Beweis.}

    Direkt nach Definition der bedingten Wahrscheinlichkeit gilt

    \[
        \pe(A\mid B)\,\pe(B)
        =
        \pe(A\cap B)
        =
        \pe(B\mid A)\,\pe(A).
    \]

    Division durch \(\pe(B)\) ergibt die gesuchte Formel:

    \[
        \pe(A\mid B)
        =
        \pe(B\mid A)\cdot\frac{\pe(A)}{\pe(B)}.
    \]
\end{satz}


\begin{bsp} \txc{(2025+2026 zsm.)} \\
Neue Frage:

    Sie haben den Wetterbericht nicht gehört, aber Sie sehen, dass Herr Zimperlich seinen Schirm dabei hat.

    Wie groß ist die Wahrscheinlichkeit, dass für diesen Tag Regen vorhergesagt wurde?

    \paragraph{Gesucht:}

    \[
        \pe(R\mid S).
    \]

    \paragraph{Problem:} Wie können wir „Bedingungen umkehren“?

    \paragraph{Lösung:} Satz von Bayes / Bayes-Formel.

    \paragraph{Lösung}

    Nach dem Satz von Bayes (einfache Version) ist

    \[
        \pe(R\mid S)
        =
        \pe(S\mid R)\cdot \frac{\pe(R)}{\pe(S)}.
    \]

    Da

    \[
        \pe(S)=1-\pe(S^c)=1-\frac{34}{100}=\frac{66}{100},
    \]

    folgt

    \[
        \pe(R\mid S)
        =
        \frac{9}{10}\cdot
        \frac{\frac{6}{10}}{\frac{66}{100}}
        =
        \frac{9}{11}.
    \]
\end{bsp}

\begin{satz}
    \textbf{Satz von Bayes (allgemeine Version)} \txc{(2025+2026 zsm.)}  \\
    Sei \((\eraum,\pe)\) ein Wahrscheinlichkeitsraum,
    \(B_1,B_2,\dots \subseteq \eraum\) Ereignisse mit

    \[
        \bigcup^+_i B_i = \eraum,
    \]

    und \(A \subseteq \eraum\) ein Ereignis mit \(\pe(A)>0\).

    Dann gilt für jedes \(i\)

    \[
        \pe(B_i\mid A)
        =
        \frac{\pe(A\mid B_i)\,\pe(B_i)}
        {
        \oub{\sum_j \pe(A\mid B_j)\,\pe(B_j)}{\overset{*}{=}\pe(A)}
        }.
    \]
    \txo{* Satz der totalen WS}

    \paragraph{Beweis.}

    Kombiniere die einfache Version des Satzes von Bayes

    \[
        \pe(B_i\mid A)
        =
        \pe(A\mid B_i)\,\frac{\pe(B_i)}{\pe(A)}
    \]

    mit der Formel von der totalen Wahrscheinlichkeit

    \[
        \pe(A)
        =
        \sum_j \pe(A\mid B_j)\,\pe(B_j).
    \]

    Einsetzen von \(\pe(A)\) in die Bayes-Formel ergibt

    \[
        \pe(B_i\mid A)
        =
        \frac{\pe(A\mid B_i)\,\pe(B_i)}
        {\sum_j \pe(A\mid B_j)\,\pe(B_j)}.
    \]

    Die Details sind eine Hausaufgabe.
\end{satz}
