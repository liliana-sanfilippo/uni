\section*{Grundgedanke}
Die Verteilung charakterisiert eine \Gls{zv} $X$ eindeutig. \\
Sind $X$ und $Y$ zwei \Glspl{zv} \un{mit derselben Verteilung}, so gilt
\[
    \pe(X \in A) = \pe_X(A) = \pe_Y(A) = \pe(Y \in A)
\]
für jedes $A$. \\
Es ist nicht immer klar, welche Verteilung zu einem sinnvollen Modell führt.
Auch sind die Parameter der Verteilung nicht unbedingt bekannt. Siehe
\begin{itemize}
    \item \textbf{Binomialverteilung:}
    \[
        \binomv(k)
        =
        \binom{n}{k}
        p^k (1-p)^{n-k},
        \qquad
        k \in \{0,\dots,n\}.
    \]

    \item \textbf{Geometrische Verteilung:}
    \[
        G_p(n)
        =
        p(1-p)^{n-1},
        \qquad
        n \in \nz.
    \]
    % TODO gloss für geom verteilung

    \item \textbf{Poisson-Verteilung:}
    \[
        P_{\lambda}(k)
        =
        \frac{\lambda^k}{k!} e^{-\lambda},
        \qquad
        k \in \nz_0.
    \]

    \item \textbf{Exponentialverteilung:}
    \[
        f(x)
        =
        \ele e^{-\lambda x}\,
        \chi_{[0,\infty)}(x),
        \qquad
        x \in \rz.
    \]
\end{itemize}
Wie können wir trotzdem Vorhersagen treffen? \\ \newline

\paragraph{Typische Fragestellungen:} Wie \dots \textit{im Mittel}? \\
\begin{itemize}
    \item Wie groß ist der mittlere Gewinn bei einem Glücksspiel?
    \item Wie lange muß man bei einem wiederholten Zufallsexperiment im Mittel

    \item Wie viele weitere Personen steckt eine infizierte Person im Mittel an?
\end{itemize}

\begin{bsp}
    \textbf{Ansteckungen}
    \begin{itemize}
        \item Die \textbf{Basisreproduktionszahl} \(R_0\) gibt an, wie viele Personen
        eine infizierte Person im Mittel ansteckt, wenn es in der Bevölkerung keine
        Immunität gibt. Gemäß RKI gilt:
        \[
            R_0 \in [2.8,\,3.8]
        \]
        (vor möglicherweise ansteckenderen Mutationen).

        \item Die \textbf{zeitabhängige Reproduktionszahl}
        \[
            R = R(t)
        \]
        berücksichtigt Effekte durch Hygienemaßnahmen, Impfungen und Immunität.
        Sie gibt an, wie viele Personen eine infizierte Person in einem Zeitfenster
        gegebener Länge im Mittel ansteckt.

        \item Die \textbf{Inzidenz} setzt Neuinfektionen in Relation zur
        Bevölkerungsgröße. Die \emph{7-Tage-Inzidenz} gibt beispielsweise an,
        wie viele neu infizierte Personen es in den vergangenen sieben Tagen
        pro \(100\,000\) Einwohner gegeben hat.
    \end{itemize}
\end{bsp}

\begin{bsp}
    \textbf{1 $\times$ Würfeln mit einem fairem Würfel} \\
    \txc{Da herrscht eine Gleichverteilung, daher ist die erwartete Augenzahl ist das arithmetische Mittel.} \\
    \[
        \eraum = \{1,\dots,6\}
    \]
    mit Gleichverteilung \(\pe\).

    Die Zufallsvariable
    \[
        X : \eraum \to \rz,
        \qquad
        X(\ele)=\ele,
    \]
    besitzt das Bild
    \[
        X(\eraum)=\{1,\dots,6\}\subseteq\rz.
    \]

    Für einen fairen Würfel ist der Erwartungswert die mittlere Augenzahl, also das
    arithmetische Mittel:
    \[
        E[X]
        =
        \frac{1+2+3+4+5+6}{6}
        =
        3.5.
    \]

    Die erwartete Augenzahl entspricht der mittleren Zahl von Schritten, die man
    in einem Würfelspiel pro Wurf vorwärts gehen würde.
\end{bsp}

\begin{bsp}
    \textbf{Spiel, bei dem nicht alle Ausgänge gleichwahrscheinlich sind} \\
    1 $\times$ Würfeln eines fairen Würfels \\
    Auszahlung:
    \begin{itemize}
        \item $k$ Euro, falls 2 gewürfelt
        \item 6 Euro, falls $2, \cdots , 6$ gewürfelt
        \item \(
                  \eraum = \{1,\dots,6\}
        \) mit Gleichverteilung \(\pe\).
    \end{itemize}
    \glssymbol{eraum} = $\kl{1, \dots, 6}$ mit Gleichverteilung \pe  \\
    Die Auszahlung wird beschrieben durch die \Gls{zv} $Y$ mit
    \[
        Y(\glssymbol{elementarereignis}) = k \cdot \chi_{\kl{1}}(\ele) + 6 \cdot \chi_{\kl{2, \dots, 6}}(\ele) \quad \faoio
    \]
    Erwartete Auszahlung als arithmetisches Mittel:
    \mab{\E{Y} = \sum_{\oio} Y(\ele) \glssymbol{wfk}(\ele) = \frac{k+ 5 \cdot 6}{6} = 6 \cdot \frac{5}{6} + k \cdot \frac{1}{6}  = 6 \cdot \pe(Y=6) + k \cdot \pe(Y=k) = \sum_{I\in Y(\glssymbol{eraum})} I \cdot \pe(Y=I)}

\end{bsp}

\begin{defi}
    Der \Gls{ew} einer \zv $X$ mit abzählbarem Bildbereich $X(\eraum) \in \rz$ ist gegeben durch
    \mab{\E{X} = \mathbb{E} X = \sum_{x\in X(\eraum)} x \cdot \pe (X=x) }
    insofern die Reihe \glsverb{abskonv}, also \ma{ \sum_{x\in X(\eraum)} |x| \cdot \pe (X=x) < \infty}. \\
    Wir sprechen in diesem Fall von der Existenz des \Gls{ew}s. \\ \newline
    Der \Gls{ew} einer \zv $X$ mit einer \dichte \glssymbol{dichte} ist gegeben durch
    \mab{\E X = \int_{- \infty}^{\infty} x \glssymbol{dichte}(x) \dx = \lim_{N \rightarrow \infty} \int_{-N}^{N} x \glssymbol{dichte}(x) \dx }
    insofern \ma{\int_{- \infty}^{\infty} |x| \glssymbol{dichte}(x) \dx < \infty}. \\
    Wir sprechen in diesem Fall von der Existenz des Erwartungswerts.
    \paragraph{Hinweis:} Für \glspl{dichte} \dcht, die gerade Funktionen sind, gilt \(\int_{-N}^N x \dcht (x) \dx = 0\) für
    alle N, auch wenn der Erwartungswert ggf. gar nicht existiert.
\end{defi}

\begin{bem}
    Absolute Konvergenz der Reihe, die den Erwartungswert definiert, garantiert,
    daß wir die Summanden beliebig umsortieren dürfen. Es gilt daher
    \[
        \txo{\E{X} \overset{Def.}{=}} \sum_{x\in X(\eraum)} x \pe(X=x) = \sum_{x\in \eraum} X(\ele) \wfk(\ele)
    \]
    sofern eine der Reihen absolut konvergent ist. (Vgl. obiges Bsp. mit dem Spiel.)

    Wenn $\eraum$ oder $X(\eraum)$ endlich ist, so existiert der Erwartungswert automatisch.

    Nimmt X nur nichtnegative Werte an, können wir einfach anfangen, den
    Erwartungswert auszurechnen, weil
    \[
        \sumxinXO  |x| \pe(X=x) = \sumxinXO x \pe(X=x)
    \]
    bzw.
    \[
        \intinfs |x| \dcht (x) \dx = \intinfs x \dcht (x) \dx \txc{ = \int^{\infty}_0 x \dcht (x) \dx }
    \]
\end{bem}

\begin{bsp}
    \textbf{Erwartungswert der Poisson-Verteilung} \\
    X Poisson-verteilt, d.h., es existiert ein $\lambda > 0$ mit
    \[
        \wfk_X(k) = \pe (X=k) = \frac{\lambda^k}{k!} e^{-\lambda} \quad \text{für alle } k \in \nz_0 = \{0, 1, 2, 3, . . . \}
    \]
    Der Erwartungswert existiert, weil die folgende Rechnung ein endliches Ergebnis hat: \\
    (Wir verwenden in dieser Argumentation, daß X nur nichtnegative Werte hat, d.h., $X \geq 0$.)
    \begin{align*}
        \E{X} & \overset{\txo{*1}}{=} \sum^{\infty}_{k=0} k \pe(X=k) \\
        & \overset{\txo{*2}}{=}  \sum^{\infty}_{k=0} \frac{\lambda^k}{k!} e^{-\lambda} \\
        & = \sum^{\infty}_{\txr{k=1}} \txg{k} \cdot \frac{\txb{\lambda^k}}{\txg{k!}} e^{-\lambda} \\
        & = \txb{\lambda}   \sum^{\infty}_{\txr{k=1}}  \frac{\txb{\lambda^{k-1}}}{\txg{(k-1)!}} e^{-\lambda} \\
        & = \lambda e^{-\lambda} \sum^{\infty}_{\txr{k=1}}  \frac{\lambda^{k-1}}{(k-1)!} \\
        & = \lambda e^{-\lambda}
        \oub{
            \sum_{l=0}^{\infty}
            \frac{\lambda^{\txr{l}}}{\txr{l}!}
        }{=e^{\lambda}} \\
        & = \lambda
    \end{align*}
    \txo{
        \begin{align*}
            *1 & \quad X(\eraum) = \nz_0 \\
            *2 & \quad \pe(X=k) = \wfk_X(k) = ...
        \end{align*}
    }
\end{bsp}

    \]

    \txo{
    \[
        h(x) = x^k \Rightarrow h'(x) = k x^{k-1}

        \E{X}&  \overset{\txo{*1}}{=}   \sum^{\infty}_{k=1} \txr{k} p(1-p)^{\txr{k-1}} \\

        & = \frac{1}{p}
    \end{align*}
    \txo{
        \begin{align*}
            *1 & \quad \text{Definition Erwartungswert } \\

    }

\begin{bsp}
    \textbf{Erwartungswert der Exponentialverteilung} \\
    \txo{
    Erinnerung:}
    \[
       \txo{ F(x) := - e^{-\alpha x}, \quad F'(x) := \alpha e^{-\alpha x}, \quad \E{X} = \frac{1}{\alpha}}
    \]

    Der Erwartungswert existiert genau dann, wenn die folgende Berechnung einen
    endlichen Wert ergibt (was der Fall ist, Wir verwenden wieder, daß $X \geq 0$):
    \begin{align*}
        \E{X} & = \int^{\infty}_0 \underbrace{x}_{\downarrow} \underbrace{\alpha e^{- \alpha x}}_{\uparrow} \dx \\
        & = - \lim_{N \rightarrow \infty} \oub{x  e^{- \alpha x}}{=0} |^N_{x=0} +  \int^{\infty}_0 \oub{1 \cdot  e^{- \alpha
        x} }{=f(x) \cdot \frac{1}{\alpha}} \dx \\
        & = 0 + \frac{1}{\alpha} \oub{\intinfs \dcht(x) \dx }{=1} \\
        & = \frac{1}{\alpha}^{\txo{*}}
    \end{align*}
    \txc{
        \begin{align*}
        * \dcht(x) & = \alpha  e^{- \alpha x} \chi_{[0,\infty)}(x) \\
            \Rightarrow \int_0^{\infty} e^{- \alpha x} \dx
            & = \frac{1}{\alpha} \int_0^{\infty} \alpha  e^{- \alpha x} \dx \\
            & = \frac{1}{\alpha}  \intinfs \dcht(x) \dx \\
            & = \frac{1}{\alpha}
        \end{align*}
    }
\end{bsp}

\begin{satz}
    \textbf{Der Erwartungswert ist linear} % TODO 13
    Geg. \glspl{zv} $X : \eraum \rightarrow \rz $ und  $Y : \eraum \rightarrow \rz $ auf $\wraum$; Konstanten $a,b \in \rz$. \\
    Annahme: $\E{X}$ und $\E{Y}$ existieren. \\
    Dann gelten $aX + bY$ ist ebenfalls eine Zufallsvariable.
    Der Erwartungswert von $aX + bY$ existiert und berechnet sich als
    \[
        \E{aX + bY} = a\E{X} + b\E{Y}
    \]
    \paragraph{Beweis (für Zufallsvariable mit abzählbarem Bildbereich)}
    Nehmen wir zunächst an, der Erwartungswert von $aX + bY$ existiert.
    Dann dürfen alle Reihen beliebig umsortiert werden:
    \begin{align*}
        \E{aX + bY} & = \sum_{\oio} [aX(\ele) + bY(\ele)] \wfk(\ele) \\
        & = a \sum_{\oio} X(\ele) \wfk(ele) + \sum_{\oio} Y(\ele) \wfk(ele) \\
        & = a\E{X} + b\E{Y}
    \end{align*}
    Eine analoge Rechnung mit Betragszeichen und Abschätzungen zeigt die
    Existenz des Erwartungswerts:
    \[
        \sum_{\oio} \oub{ | aX(\ele) + bY(\ele)| }{\leq |a| |X(\ele)| + |b| |Y(\ele)|} \wfk(\ele) \leq |a| \oub{\sum_{\oio} |X(\ele)| \wfk(\ele)}{< \infty\ unab.\ Ver.} + |b|  \oub{\sum_{\oio} |Y(\ele)| \wfk(\ele)}{< \infty\ unab.\ Ver.} < \infty
    \]
    da nach Voraussetzung $\E{X}$ und $\E{Y}$ existieren.
\end{satz}

\begin{bsp}
    \textbf{Augensumme bei mehrfachem Würfeln mit fairem Würfel}
    \[
        \eraum = \{1,\dots,6\}^n
    \]
    mit Gleichverteilung \(P\).

    Für \(i=1,\dots,n\) sei
    \[
        X_i(\ele)=\ele_i,
    \]
    wobei \(\ele=(\ele_1,\dots,\ele_n)\in\eraum\).

    Die Zufallsvariable
    \[
        X=\sum_{i=1}^{n} X_i
    \]
    beschreibt die Augensumme.

    \textbf{Berechnung der erwarteten Augensumme für zweimaliges Würfeln (\(n=2\)):}

    Direkt über die Verteilung:
    \[
        E[X]
        =
        \sum_{i=2}^{12}
        i \, \pe(X_1+X_2=i) = ...
    \]
    was etwas aufwendig ist.

    Alternativ:
    \[
        \begin{aligned}
            E[X]
            &=
            \sum_{\ele\in\eraum}
            \bigl(X_1(\ele)+X_2(\ele)\bigr)\,\wfk(\ele) \\
            &=
            \frac{1}{36}
            \sum_{\ele_1, \ele_2=1}^{6} [\ele_1 + \ele_2] \\
            &=
            \frac{1}{36}
            \sum_{\ele_1=1}^{6}
            \sum_{\ele_2=1}^{6}
            (\ele_1+\ele_2) \\
            &=
            \frac{1}{36}
            \left[
                6\sum_{\ele_1=1}^{6}\ele_1
            +
            6\sum_{\ele_2=1}^{6}\ele_2
            \right] \\
            &=
            \frac{1}{36}
            \left(
                6\cdot 21 + 6\cdot 21
            \right)
            =
            7.
        \end{aligned}
    \]

    Mit der Linearität des Erwartungswertes erhält man sofort:
    \[
        E[X]
        =
        E[X_1]+E[X_2]
        =
        2\cdot 3,5
        =
        7.
    \]

    Dies verallgemeinert sich unmittelbar auf \(n\)-maliges Würfeln:
    \[
        E[X]
        =
        E[X_1+\cdots+X_n]
        =
        E[X_1]+\cdots+E[X_n]
        =
        n\cdot\frac{7}{2}.
    \]
\end{bsp}

\begin{bsp}
    \textbf{Binomialverteilung} \txo{$\E{X} = n \cdot p$} \\
    \txo{
    $X(\eraum) = \{ 0,...,n\}, \quad \# X(\eraum) = n+1 < \infty$}
    \begin{itemize}
        \item \(X\) sei binomialverteilt mit den Parametern \(n\) und \(p\).
        \item Der Erwartungswert$^{\txo{*}}$ von \(X\) existiert, da der Bildbereich von \(X\)
        endlich ist.
    \end{itemize}
    \txo{erwartete Anzahl an Erfolgen in einem n-fach unabhängig wiederholten Bernoulli-Experiment}

    \paragraph{1. Versuch} (mit binomischer Formel und etwas Geschick)
    \[
        \E{X} = \sum^n_{k=0} k \oub{\binom{n}{k} p^k (1-p)^{n-k}}{=\pe(X=x)} = ...
    \]

    \paragraph{2. Versuch}
    Wähle
    \[
        \eraum=\{0,1\}^n.
    \]

    Die Anzahl der Erfolge lässt sich darstellen als
    \[
        X=\sum_{i=1}^{n} X_i,
    \]
    wobei
    \[
        X_i(\ele)=\ele_i.
    \]

    Für jede Indikatorvariable gilt
    \[
        E[X_i]
        =
        0\cdot \pe(X_i=0)
        +
        1\cdot \pe(X_i=1)
        =
        p.
    \]

    Mit der Linearität des Erwartungswertes folgt unmittelbar:
    \[
        E[X]
        =
        \sum_{i=1}^{n} \oub{E[X_i]}{=p}
        =
        \sum_{i=1}^{n} p
        =
        np.
    \]
\end{bsp}

\begin{bsp}
    \textbf{St. Petersburger Paradoxon (Erwartungswert muß nicht existieren)} \\
    Ein Casino möchte folgendes Spiel anbieten:
    Eine faire Münze wird geworfen, bis zum ersten Mal Kopf erscheint.
    Passiert dies beim n-ten Wurf, so sei die Auszahlung $X = 2^n$
    \[
        \eraum=\nz,
        \qquad
        \wfk(n)=p(1-p)^{n-1},
        \qquad
        p=\frac12.
    \]
    Die Zufallsvariable $X:\eraum\to\rz$ sei definiert durch $ X(n)=2^n$ \txo{(Ausazhlung)} für alle $n\in\eraum$

    \[
        \wfk_X(2^n)
        =
        \pe( \oub{X=2^n}{\text{Kopf erstmalig im n-ten Wurf}} )
        =
        \wfk(n)
        =
        p(1-p)^{n-1}
        =
        2^{-n},
        \qquad
        n\in \eraum = \nz.
    \]
    \txo{$\wfk(n) = (\frac{1}{2})^n = 2^{-n} \quad \forall n \in \eraum = \nz$} \\
    Was wäre ein fairer Einsatz für dieses Spiel?
    (Wir nennen den Einsatz fair, wenn der Einsatz gleich der erwarteten Auszahlung ist.) \\
    Wir suchen also $\E{X}$

    \txo{$X \geq 0$, EW existiert genau dann, wenn Rechnung endliches Ergebnis hat}

    Nun ist jedoch
    \[
        \sum_{n=1}^{\infty} 2^n \pe(X=2^n) = \sum_{n=1}^{\infty} 2^n \wfk_X(2^n) = \sum_{n=1}^{\infty} 2^n 2^{-n} = \sum_{n=1}^{\infty} 1 = \infty
    \]
    Der Erwartungswert existiert also nicht.
    Ein fairer Einsatz kann nicht bestimmt werden.
\end{bsp}

\begin{satz}
    \textbf{Satz (P)} \\
    Seien
    \[
        X:\Omega\to\mathbb{R}
        \quad\text{und}\quad
        Y:\Omega\to\mathbb{R}
    \]
    zwei unabhängige Zufallsvariablen, deren Erwartungswerte existieren.

    Dann existiert auch der Erwartungswert von \(X\cdot Y\), und es gilt
    \[
        \E{X\cdot Y}
        =
        \E{X}\cdot\E{Y}.
    \]

    \paragraph{Beweis.}
    \paragraph{Schritt 1:}
    Wir nehmen zunächst an, dass der Erwartungswert von \(X\cdot Y\) existiert.

    Wir beginnen mit der folgenden Darstellung. Für \(z\neq 0\) gilt: \txo{(xy = z)}
    \[
        \begin{aligned}
            P(XY=z)
            & \overset{\txo{*}}{=}
            \sum_{y\neq 0}
            P(XY=z\mid Y=y)\,P(Y=y)
            \\
            &=
            \sum_{y\neq 0}
            P\left(X=\frac{z}{y}\mid Y=y\right)
            P(Y=y)
            \\
            &=
            \sum_{y\neq 0}
            P\left(X=\frac{z}{y}\right)
            P(Y=y),
        \end{aligned}
    \]
    da \(X\) und \(Y\) unabhängig sind.
    \[
        \txo{
            \pe(X \cdot X = z) = \sum_{y \in Y(\eraum)} \pe(X \cdot Y = z, Y = y) \overset{z \neq 0}{=} \sum_{y \in Y(\eraum), y \neq 0} \pe(X \cdot Y = z| Y = y) \cdot \pe(Y=y)
        }
    \]
    Aus
    \[
        P(XY=z)
        =
        \sum_{y\neq 0}
        P\left(X=\frac{z}{y}\right)
        P(Y=y),
        \qquad z\neq 0,
    \]
    folgt
    \[
        \begin{aligned}
            \E{XY}
            &=
            \sum_{z\in XY(\Omega)\txo{, z \neq 0}}
            z\,P(XY=z)
            \\
            &=
            \sum_{z\neq 0}
            \sum_{y\neq 0}
            \frac{z}{y}\,y\,
            P\left(X=\frac{z}{y}\right)
            P(Y=y)
            &=
            \sum_{y\neq 0}
            \sum_{x\neq 0}
            xy \pe(X=x) \pe(Y=y)
            &=
            \big(
            \sum_{y\neq 0} y \pe(Y=y)
            \big)
            \big(
            \sum_{x\neq 0} x \pe(X=x)
            \big)
            &=
            \E{X} \cdot \E{Y}
        \end{aligned}
    \]
    \paragraph{Schritt 2:} Eine analoge Rechnung zeigt die Existenz des Erwartungswerts:
\end{satz}
\begin{satz}
    \textbf{Kurzfassung Satz (P)}
    \[
        X,Y \text{ unabhängig mit ex. EW}
        \;\Longrightarrow\;
        \E{XY}
        =
        \E{X}\,\E{Y}.
    \]
\end{satz}

\begin{bsp}
    \[
        \E{X\cdot Y}
        =
        \E{X}\cdot\E{Y}
    \]
    impliziert nicht, dass \(X\) und \(Y\) unabhängig sind.

    \begin{itemize}
        \item Definiere Hilfszufallsvariablen \(X_1\) und \(Y_1\), unabhängig mit
        \[
            P(X_1=0)=P(X_1=1)=\frac12
        \]
        und
        \[
            P(Y_1=0)=P(Y_1=1)=\frac12.
        \]

        \item Setze
        \[
            X:=X_1+Y_1,
            \qquad
            Y:=|X_1-Y_1|.
        \]
    \end{itemize}

    \paragraph{Beobachtungen}

    \begin{itemize}
        \item Bildbereiche:
        \[
            X(\Omega)=\{0,1,2\},
            \qquad
            Y(\Omega)=\{0,1\}.
        \]

        \item
        \[
            Y=0
            \iff
            X_1=Y_1
            \iff
            X\in\{0,2\}.
        \]

        \item
        \[
            Y=1
            \iff
            X_1\neq Y_1
            \iff
            X=1.
        \]
    \end{itemize}

    \paragraph{\(X\) und \(Y\) sind nicht unabhängig.}

    \[
        P(X=1,Y=0)=0,
    \]
    da
    \[
        \{X=1,Y=0\}=\varnothing.
    \]

    Andererseits gilt
    \[
        P(X=1)
        =
        2\cdot\frac14
        >
        0
    \]
    und
    \[
        P(Y=0)
        =
        2\cdot\frac14
        >
        0.
    \]

    \paragraph{Dennoch gilt \(\E{XY}=\E{X}\E{Y}\).}

    \[
        \begin{aligned}
            \E{XY}
            &=
            0\cdot P(XY=0)
            +
            1\cdot P(XY=1)
            +
            2\cdot P(XY=2)
            \\
            &=
            0+P(X=1)+0
            \\
            &=
            \frac12.
        \end{aligned}
    \]

    Außerdem
    \[
        \begin{aligned}
            \E{X}\,\E{Y}
            &=
            (\E{X_1}+\E{Y_1})\,P(Y=1)
            \\
            &=
            2\,\E{X_1}\,P(X=1)
            \\
            &=
            2\cdot\frac12\cdot\frac12
            \\
            &=
            \frac12.
        \end{aligned}
    \]
\end{bsp}

\begin{bem}
    Der Satz kann nicht nur zum Berechnen von Erwartungswerten von Produkten
    unabh¨angiger Zufallsvariablen verwendet werden, sondern kann auch benutzt
    werden, um zu zeigen, daß zwei Zufallsvariablen nicht unabh¨angig sein k¨onnen,
    wenn der Erwartungswert des Produkts der Zufallsvariablen nicht ins Produkt der
    einzelnen Erwartungswerte zerf¨allt.
\end{bem}

\begin{defi}
    \textbf{Zusammengesetzte Zufallsvariablen} \\
    Eine Zufallsvariable $X: \eraum \rightarrow \rz$ und eine \txo{messbare} Funktion $f: \rz \rightarrow \rz$ können
    zusammengesetzt werden zu einer neuen Zufallsvariablen:
    \[
        Y(\ele) = (f \circ X)(\ele) = f(X(\ele))
    \]
    \txo{
    $Y(\ele) = f(X(\ele)) \quad \forall \ele$
    }
\end{defi}

\begin{satz}
    Sofern der Erwartungswert $\E{Y}$ existiert, gilt
    \[
        \E{Y} = \sum_{x \in X(\eraum)} f(x) \pe(X=x)
    \]
    \paragraph{Beweis.}
    \begin{align*}
        \E{Y} & =   \sum_{y \in Y(\eraum)} y \pe(Y=y) \\
        & =   \sum_{y \in Y(\eraum)} y  \sum_{x \in X(\eraum): f(x) = y\txo{*}} \pe(X=x) \\
        & =   \sum_{y \in Y(\eraum)}   \sum_{x \in X(\eraum): f(x) = y} \oub{y}{=f(x)} \pe(X=x) \\
        & =\sum_{y \in Y(\eraum)}   \sum_{x \in X(\eraum): f(x) = y} f(x) \pe(X=x) \\
        & = \sum_{x \in X(\eraum)} f(x) \pe(X=x)
    \end{align*}

        \[
           \txo{*  Y=y \Leftrightarrow f(x) = y, \quad \pe(Y=y) = \sum_{x \in X(\eraum): f(x) = y} \pe(X=x)}
        \]

\end{satz}

\begin{defi}
    \textbf{n-tes Moment von X} \\
    Für $n \in \nz $ heißt $\E{X^n}$ das erste n-te Moment von $X$, insofern $\E{|X|^n} < \infty$.
    (Erwartungswert ist das 1. Moment)
\end{defi}
\begin{defi}
    $\E{ ( X -  \E{X} )^n } $ heißt das n-te zentrierte Moment von $X$.
    Es existiert, wenn $\E{|X|^n} < \infty$. \\
\end{defi}
\begin{defi}
    Die Varianz von $X$ ist das 2 zentrierte Moment, d.h.
    \[
        Var(X) = \E{ ( X -  \E{X} )^2 }
    \]
    (mittlere quadratische Abweichung) \\
    Die Varianz von $X$ existiert, wenn $Var(X) < \infty$
\end{defi}

\begin{bsp}
    \textbf{Bsp.}

    Seien \(X, Y\) Zufallsvariablen mit
    \[
        \pe(X=+1)=\frac{1}{2}=\pe(X=-1)
    \]
    und
    \[
        Y=100\cdot X.
    \]

    \textbf{Anhand des Erwartungswertes: kein Unterschied!}

    \[
        \E{X}
        =1\cdot \pe(X=1)+(-1)\cdot \pe(X=-1)
        =1\cdot\frac{1}{2}-1\cdot\frac{1}{2}
        =0.
    \]

    Da \(Y=100X\) gilt,

    \[
        \E{Y}=100\cdot \E{X}=0.
    \]
    \textbf{Die Varianz zeigt den Unterschied:}

    Da \(\E{X}=0\),

    \[
        \operatorname{Var}(X)
        =\E{(X- \oub{\E{X}}{=0} )^{2}}
        =\E{ \oub{X^{2}}{1} }
        = \E{1}
        = 1
    \]

    Wegen \(X\in\{-1,1\}\) ist

    \[
        X^{2}=1,
    \]

    also

    \[
        \operatorname{Var}(X)
        =\E{1}
        =1.
    \]

    Ebenso gilt wegen \(\E{Y}=0\),

    \[
        \operatorname{Var}(Y)
        =\E{(Y-\E{Y})^{2}}
        =\E{Y^2}
        =\E{(100X)^{2}}
    \]

    Damit folgt

    \[
        \operatorname{Var}(Y)
        =\E{100^{2}X^{2}}
        =100^{2}\E{X^{2}}
        =100^{2}.
    \]
\end{bsp}
