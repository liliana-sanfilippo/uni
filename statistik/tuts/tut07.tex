\section{PÜ 1.6.4.}\label{sec:pu-1.6.4.}
Auf einem Tisch stehen zwei Urnen.
\begin{itemize}
    \item Urne $U_K$ enthalte eine schwarze Kugel und zwei weiße Kugeln.
    \item Urne $U_Z$ enthalte zwei schwarze Kugeln und eine weiße Kugel.
\end{itemize}
Die Kugeln unterscheiden sich nur durch ihre Farbe.
Mavi wirft eine faire Münze, um sich zwischen den beiden Urnen zu entscheiden.
\begin{itemize}
    \item Wenn die Münze Kopf zeigt, dann zieht Mavi nacheinander und ohne Zurücklegen zwei
    Kugeln aus Urne $U_K$.
    \item Wenn die Münze Zahl zeigt, dann zieht Mavi nacheinander und ohne Zurücklegen zwei
    Kugeln aus Urne $U_Z$.
\end{itemize}
\begin{auf}
    \subsection*{(a) Geben Sie einen geeigneten Ereignisraum $\Omega$ an, mit dem Sie die folgenden Aufgabenteile
    bearbeiten können.}
    \paragraph{Meine Lösung:} $\Omega = \{k,z\} \times \{s, w\}^2 $, wobei $\omega = (k, s, w)$ bedeutet, dass aus
    der Urne $U_K$ erst eine schwarze und dann eine weiße Kugel gezogen wurde.
    \paragraph{Tutor Lösung:} $\Omega = \{k,z\} \times \{s, w\}^2$ mit der Bemerkung, dass $\pe((k,s,s)) = \pe(z,w,w) = 0$.
    Die Bemerkung muss nicht dazu.
    Dabei ist $\{k,z\}$ der Münzwurf und $\{s, w\}^2$ das Ziehen zweier Kugeln.

    Man kann es auch direkt mit Tripeln machen, aber das ist mehr Schreibarbeit.
    Aber dann hat man die Sachen mit $\pe = 0$ auch nicht drin, sondern nur das, was tatsächlich möglich ist.
\end{auf}

\begin{auf}
    \subsubsection*{(b) Definieren Sie jene Ereignisse als Teilmengen von $\Omega$, die Sie benötigen, um die folgenden
    Aufgabenteile bearbeiten zu können.}
    \paragraph{Meine Lösung:}
    \begin{align*}
        S_i & = \{\text{i schwarze Kugeln gezogen}\} \\
        W_i  & = \{\text{i weiße Kugeln gezogen}\} \\
        S_1 = W_1 & = \{(k,s,w), (k,w,s), (z,w,s), (z,s,w)\} \\
        S_0 = W_2 &  = \{(k,w,w)\} \\
        S_2  = W_0 &  = \{(z,s,s)\} \\
        U_K & = \{\text{Es wurde aus Urne $U_K$ gezogen}\} \\
        & = \{(k,s,w), (k,w,s), (k,w,w)\}  \\
        U_Z & = \{\text{Es wurde aus Urne $U_Z$ gezogen}\} \\
        & = \{(z,s,w), (z,w,s), (z,s,s)\}
    \end{align*}
    (Alle Ereignisse, nicht nur die für die Aufgaben relevanten\ldots)
    \paragraph{Tutor Lösung:}
    \begin{align*}
    W &= \{k,z\} \times \{w\}^2 \\
    K &= \{k\} \times \{s,w\}^2 \\
    Z &= \{z\} \times \{s,w\}^2
    \end{align*}
\end{auf}

\begin{auf}
    \subsubsection*{(c) Bestimmen Sie aus dem Text die Wahrscheinlichkeiten bzw. bedingten Wahrscheinlichkeiten
    der Ereignisse aus (b).}
    \paragraph{Meine Lösung:}
    \begin{align*}
        P(U_Z) = P(U_K) & = \frac{1}{2} \\
        P(S_0 \mid U_K) & = \frac{2}{3} \cdot \frac{1}{2} = \frac{2}{6} \\
        P(S_0 \mid U_K) & = 0
    \end{align*}
    (Nicht weiter gemacht)
    \paragraph{Tutor Lösung:}
    \begin{align*}
        \pe(K) & = \pe(Z) = \frac{1}{2} \\
        \pe(W \mid K) & = \frac{2}{3} \cdot \frac{1}{2} = \frac{2}{6} = \frac{1}{3} \\
        \pe(W \mid Z) & = 0 \\
    \end{align*}
\end{auf}

\begin{auf}
    \subsubsection*{(d) Berechnen Sie die Wahrscheinlichkeit, dass Mavi mindestens eine schwarze Kugel zieht
        (d.h., die Wahrscheinlichkeit, dass Mavi eine oder zwei schwarze Kugeln zieht)}
    \paragraph{Tutor Lösung:} $\pe(\text{``Mindestens eine schwarze Kugel gezogen''}) = \pe(W^c) = 1 - \pe(W)$, wobei
    $W = \{\text{``Es werden nur weiße Kugeln gezogen''}\}$.
    \begin{align*}
        \pe(W) & = \pe(W \mid K) \cdot \pe(K) + \pe(W \mid Z) \cdot \pe(Z) \\
        & = \frac{1}{3} \cdot \frac{1}{2} + 0 \cdot \frac{1}{2} \\
        & = \frac{1}{6}
    \end{align*}
\end{auf}

\begin{auf}
    \subsubsection*{(e) Sie wissen, dass Mavi mindestens eine schwarze Kugel gezogen hat. Berechnen Sie die
    Wahrscheinlichkeit, dass Mavi aus der Urne $U_Z$ gezogen hat.}
    \paragraph{Tutor Lösung:}
    \begin{align*}
        \pe(\text{``Zahl''} \mid \text{``min. 1 schwarze Kugel''}) & = \pe(Z \mid W^c) \\
        & \overset{Bayes}{=} \frac{\pe(W^c \mid Z) \cdot \pe(Z)}{\pe(W^c)} \\
        & = \frac{(1-\pe(W \mid Z)) \cdot \pe(Z)}{\pe(W^c)} \\
        & = \frac{\frac{1}{2}}{\frac{5}{6}} = \frac{6}{10} \\
        & = \frac{3}{5}
    \end{align*}
\end{auf}
